         function showRoomSlides() {
           let i;
           let rSlides = document.getElementsByClassName("roomSlides");
           for (i = 0; i < rSlides.length; i++) {
             rSlides[i].style.display = "none";  
           }
           roomIndex++;
           if (roomIndex > rSlides.length){roomIndex = 1}    
          
           rSlides[roomIndex-1].style.display = "block";
           setTimeout(showRoomSlides, 5000);
        }

        function showRoom(n) {

            
            let i;
            let nextSlides = document.getElementsByClassName("roomSlides");

            if (n > nextSlides.length) {nextRoomIndex = 1}    
            if (n < 1) {nextRoomIndex = nextSlides.length}
            for (i = 0; i < nextSlides.length; i++) {
                nextSlides[i].style.display = "none";  
            }
            nextSlides[nextRoomIndex-1].style.display = "block";
            
            setTimeout(showRoom, 3000);
        }
